# stylelint 在 vue 中的配置

# 安装

npm install stylelint --save-dev npm install stylelint-config-recommended --save-dev npm install stylelint-config-standard --save-dev

# 新建.stylelintrc.js/.stylelintrc.json/.stylelintrc 任意一个文件

eslint 官网

module.exports = {
  extends: ['stylelint-config-recommended', 'stylelint-config-standard'],
  rules: {
    // 规则
    indentation: 2, // 首行缩进2个字符
    'rule-empty-line-before': 'never', // 是否有空行
    'string-quotes': 'single', // 使用单引号，可以为double
    'selector-list-comma-newline-after': 'never-multi-line', // 多个选择器之间是否换行
    'selector-pseudo-element-colon-notation': 'single', // 选择器伪元素冒号表示法单冒号，可以为double
    'no-descending-specificity': null, // 禁止低优先级的选择器出现在高优先级的选择器之后
    'media-query-list-comma-newline-before': 'never-multi-line', // 在媒体查询的逗号之前禁止有一换行
    'no-empty-source': null, // 禁止空源
    'no-missing-end-of-source-newline': null, // 禁止缺少文件末尾的换行符
    'number-leading-zero': 'always', // 禁止小于 1 的小数的前导 0,always为0.8，never为.8
    'function-url-quotes': 'always', // url使用引号,可以为never，比如a { background: url('x.jpg') }
    'font-weight-notation': 'numeric', // 使用数字或命名的 (可能的情况下) font-weight 值
    'at-rule-no-unknown': [
      true,
      {
        ignoreAtRules: ['studio']
      }
    ]
  }
}

# vue.config.js 配置: 自动修复、自动检测

npm install --save-dev stylelint-webpack-plugin

chainWebpack: (config) => {
  config.plugin('stylelint-webpack-plugin').use('stylelint-webpack-plugin', [
    {
      // files: ['**/*.{html,vue, styl}'], // 运行时自动检测代码css样式规范，可以单独设置某些文件需要
      files: ['src/views/studio/info.vue'],
      fix: true, // 是否自动修复
      cache: true, // 缓冲
      emitErrors: true, // 是否抛出错误
      failOnError: false // 报错时是否编译失败
    }
  ])
}

# git 提交时自动检测

在 package.json 中配置:

{
    ...
    "lint-staged": {
        "*.{vue,js}": [
            "eslint --fix",
            "git add"
        ],
        "*.{html,vue,css,sass,scss,style}": [ // 配置需要修复的文件后缀名
            "stylelint --fix",
            "git add",
        ]
    },
    "husky": {
        "hooks": {
            "pre-commit": "lint-staged",
        }
    },
    "scripts": { // 配置命令
        "lint:style": "stylelint --config .stylelintrc.js src/**/*.vue --syntax stylus",
    }
}

# vscode 配置

  1. 安装 vscode 插件: stylelint
  2. setting.json 配置
"css.validate": true,
"less.validate": true,
"scss.validate": true,
"editor.codeActionsOnSave": {
    "source.fixAll.eslint": true, // 使用eslint修复
    "source.fixAll.stylelint": true // 使用stylelint修复
},
"stylusSupremacy.insertColons": true,
"stylusSupremacy.insertSemicolons": true, // 使用分号
"stylusSupremacy.insertBraces": true,// 使用大括号
"stylusSupremacy.insertNewLineAroundImports": false, // import时不插入新行
"stylusSupremacy.insertNewLineAroundBlocks": false, // block之间不加空行
